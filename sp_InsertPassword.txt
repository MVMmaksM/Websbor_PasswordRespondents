CREATE PROCEDURE [dbo].[sp_InsertPassword]
    @name nvarchar(100),
    @okpo nvarchar(15),
	@password nvarchar(15),
	@datecreate nvarchar(15),
    @comment nvarchar(100),
    @ID int out
AS
    INSERT INTO Password (name, okpo, password, datecreate, comment)
    VALUES (@name, @okpo, @password, @datecreate, @comment)
   
    SET @ID=SCOPE_IDENTITY()
